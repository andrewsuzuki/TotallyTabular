00120 inline StkFloat Envelope :: tick( void )
00121 {
00122   if ( state_ ) {
00123     if ( target_ > value_ ) {
00124       value_ += rate_;
00125       if ( value_ >= target_ ) {
00126         value_ = target_;
00127         state_ = 0;
00128       }
00129     }
00130     else {
00131       value_ -= rate_;
00132       if ( value_ <= target_ ) {
00133         value_ = target_;
00134         state_ = 0;
00135       }
00136     }
00137     lastFrame_[0] = value_;
00138   }
00139 
00140   return value_;
00141 }
